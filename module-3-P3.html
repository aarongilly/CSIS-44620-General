# Generate a sequence of 5 random integers
random_numbers = np.random.randint(1, 100, size=5)
x = np.arange(len(random_numbers))

### Using Matplotlib to create a scatter plot and a line plot ###

fig, axs = plt.subplots(1, 2, figsize=(10, 4))

# Scatter plot
axs[0].scatter(x, random_numbers) # scatter method - generates the scatter plot
axs[0].set_title("Scatter Plot of Random Numbers")
axs[0].set_xlabel("Index")
axs[0].set_ylabel("Value")

# Line plot
axs[1].plot(x, random_numbers) # plot method - generates the line plot
axs[1].set_title("Line Plot of Random Numbers")
axs[1].set_xlabel("Index")
axs[1].set_ylabel("Value")

plt.tight_layout()
plt.show()
